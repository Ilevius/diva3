{\LARGE Интеграл}
\[
\mathbf{u}_{+}\left(  x,z\right)  =\frac{1}{2\pi}\int\limits_{\Gamma
}\mathbf{U}_{+}\left(  \alpha,z\right)  e^{-i\alpha x}d\alpha,\,z\leq-h
\]%
\[
\mathbf{U}_{+}=\left\{
\begin{array}
[c]{l}%
U_{+}=t_{3}\left(  -i\alpha\right)  e^{s_{1}\left(  z+h\right)  }-t_{4}%
s_{2}e^{s_{2}\left(  z+h\right)  }\\
W_{+}=t_{3}s_{1}e^{s_{1}\left(  z+h\right)  }+t_{4}\left(  -i\alpha\right)
e^{s_{2}\left(  z+h\right)  },
\end{array}
\right.
\]%
\[
\mathbf{U}_{+}=\left(
\begin{array}
[c]{l}%
-i\alpha\\
s_{1}%
\end{array}
\right)  t_{1}^{3}e^{s_{1}\left(  h+z\right)  -h\sigma_{1}}+\left(
\begin{array}
[c]{l}%
-i\alpha\\
s_{1}%
\end{array}
\right)  t_{2}^{3}e^{s_{1}\left(  h+z\right)  -h\sigma_{2}}+\left(
\begin{array}
[c]{l}%
-s_{2}\\
-i\alpha
\end{array}
\right)  t_{1}^{4}e^{s_{2}\left(  h+z\right)  -h\sigma_{1}}+\left(
\begin{array}
[c]{l}%
-s_{2}\\
-i\alpha
\end{array}
\right)  t_{2}^{4}e^{s_{2}\left(  h+z\right)  -h\sigma_{2}}=
\]%
\[
=\mathbf{U}_{pp}^{+}+\mathbf{U}_{ps}^{+}+\mathbf{U}_{sp}^{+}+\mathbf{U}%
_{ss}^{+}.
\]


{\LARGE Асимптотика}%


\bigskip
\end{center}

Введем
замену
\[
\left\{
\begin{array}
[c]{l}%
x=R\cos\psi\\
z=R\sin\psi-2h
\end{array}
\right.
\]


\bigskip



\begin{center}
${\LARGE \eta}_{11}$
\end{center}

Пусть $\left\vert \alpha\right\vert
<\min\{\kappa_{1},\varkappa_{1}\},$ тогда
\[
s_{1}\left(  h+z\right)  -h\sigma_{1}=
\]%
\[
\eta_{11}=\left(  \hat{\sigma}_{1}+\hat{s}_{1}\right)  \frac{h}{R}-\hat{s}%
_{1}\sin\psi-\alpha\cos\psi=\left(  \sqrt{\varkappa_{1}^{2}-\alpha^{2}}%
+\sqrt{\kappa_{1}^{2}-\alpha^{2}}\right)  \frac{h}{R}-\sqrt{\kappa_{1}%
^{2}-\alpha^{2}}\sin\psi-\alpha\cos\psi
\]%
\[
\eta_{11}^{^{\prime}}=\frac{\partial}{\partial\alpha}\left(  \left(
\sqrt{\varkappa_{1}^{2}-\alpha^{2}}+\sqrt{\kappa_{1}^{2}-\alpha^{2}}\right)
\frac{h}{R}-\sqrt{\kappa_{1}^{2}-\alpha^{2}}\sin\psi-\alpha\cos\psi\right)
\]
поиск нулей
\[
h\alpha\sqrt{\kappa_{1}^{2}-\alpha^{2}}+h\alpha\sqrt{\varkappa_{1}^{2}%
-\alpha^{2}}-R\alpha\left(  \sin\psi\right)  \sqrt{\varkappa_{1}^{2}%
-\alpha^{2}}+R\left(  \cos\psi\right)  \sqrt{\kappa_{1}^{2}-\alpha^{2}}%
\sqrt{\varkappa_{1}^{2}-\alpha^{2}}%
\]%
\[
\eta_{11}^{^{^{\prime\prime}}}=\frac{\partial}{\partial\alpha}\left(
\alpha\frac{\sin\psi}{\sqrt{\kappa_{1}^{2}-\alpha^{2}}}-\cos\psi-\frac{1}%
{R}h\frac{\alpha}{\sqrt{\kappa_{1}^{2}-\alpha^{2}}}-\frac{1}{R}h\allowbreak
\frac{\alpha}{\sqrt{\varkappa_{1}^{2}-\alpha^{2}}}\right)  =
\]%
\[
=\kappa_{1}^{2}\frac{\sin\psi}{\left(  \kappa_{1}^{2}-\alpha^{2}\right)
^{\frac{3}{2}}}-\frac{h}{R}\frac{\kappa_{1}^{2}}{\left(  \kappa_{1}^{2}%
-\alpha^{2}\right)  ^{\frac{3}{2}}}-\frac{h}{R}\frac{\varkappa_{1}^{2}%
}{\left(  \varkappa_{1}^{2}-\alpha^{2}\right)  ^{\frac{3}{2}}}%
\]%
\[
\mathbf{u}_{pp}^{+}\simeq\sqrt{\frac{1}{2\pi R\left\vert \eta_{11}%
^{^{^{\prime\prime}}}\left(  \alpha_{0}\right)  \right\vert }}f(\alpha
_{0})\exp\left(  iR\eta_{11}\left(  \alpha_{0}\right)  +\frac{i\pi}{4}%
sgn\eta_{11}^{^{^{\prime\prime}}}\left(  \alpha_{0}\right)  \right)
\]


\bigskip



\begin{center}
${\LARGE \eta}_{12}$
(экспонента
$e^{s_{1}\left(  h+z\right)  -h\sigma_{2}}$ )
\end{center}

Пусть $\left\vert \alpha\right\vert
<\min\{\kappa_{1},\varkappa_{1}\},$ тогда
\[
s_{1}\left(  h+z\right)  -h\sigma_{2}-i\alpha x=s_{1}\left(  R\sin
\psi-h\right)  -h\sigma_{2}-i\alpha R\cos\psi=s_{1}R\sin\psi-s_{1}%
h-h\sigma_{2}-i\alpha R\cos\psi=
\]%
\[
=-i\widehat{s}_{1}R\sin\psi+i\widehat{s}_{1}h+ih\widehat{\sigma}_{2}-i\alpha
R\cos\psi=iR\left(  -\widehat{s}_{1}\sin\psi+\widehat{s}_{1}\frac{h}{R}%
+\frac{h}{R}\widehat{\sigma}_{2}-\alpha\cos\psi\right)
\]%
\[
\eta_{12}=\left(  \hat{\sigma}_{2}+\hat{s}_{1}\right)  \frac{h}{R}-\hat{s}%
_{1}\sin\psi-\alpha\cos\psi=\left(  \sqrt{\varkappa_{2}^{2}-\alpha^{2}}%
+\sqrt{\kappa_{1}^{2}-\alpha^{2}}\right)  \frac{h}{R}-\sqrt{\kappa_{1}%
^{2}-\alpha^{2}}\sin\psi-\alpha\cos\psi
\]%
\[
\eta_{11}^{^{\prime}}=\frac{\partial}{\partial\alpha}\left(  \left(
\sqrt{\varkappa_{2}^{2}-\alpha^{2}}+\sqrt{\kappa_{1}^{2}-\alpha^{2}}\right)
\frac{h}{R}-\sqrt{\kappa_{1}^{2}-\alpha^{2}}\sin\psi-\alpha\cos\psi\right)
\]
поиск нулей
\[
h\alpha\sqrt{\kappa_{1}^{2}-\alpha^{2}}+h\alpha\sqrt{\varkappa_{2}^{2}%
-\alpha^{2}}-R\alpha\left(  \sin\psi\right)  \sqrt{\varkappa_{2}^{2}%
-\alpha^{2}}+R\left(  \cos\psi\right)  \sqrt{\kappa_{1}^{2}-\alpha^{2}}%
\sqrt{\varkappa_{2}^{2}-\alpha^{2}}%
\]%
\[
\eta_{11}^{^{^{\prime\prime}}}=\frac{\partial}{\partial\alpha}\left(
\alpha\frac{\sin\psi}{\sqrt{\kappa_{1}^{2}-\alpha^{2}}}-\cos\psi-\frac{1}%
{R}h\frac{\alpha}{\sqrt{\kappa_{1}^{2}-\alpha^{2}}}-\frac{1}{R}h\allowbreak
\frac{\alpha}{\sqrt{\varkappa_{2}^{2}-\alpha^{2}}}\right)  =
\]%
\[
=\kappa_{1}^{2}\frac{\sin\psi}{\left(  \kappa_{1}^{2}-\alpha^{2}\right)
^{\frac{3}{2}}}-\frac{h}{R}\frac{\kappa_{1}^{2}}{\left(  \kappa_{1}^{2}%
-\alpha^{2}\right)  ^{\frac{3}{2}}}-\frac{h}{R}\frac{\varkappa_{2}^{2}%
}{\left(  \varkappa_{2}^{2}-\alpha^{2}\right)  ^{\frac{3}{2}}}%
\]%
\[
\mathbf{u}_{pp}^{+}\simeq\sqrt{\frac{1}{2\pi R\left\vert \eta_{11}%
^{^{^{\prime\prime}}}\left(  \alpha_{0}\right)  \right\vert }}f(\alpha
_{0})\exp\left(  iR\eta_{11}\left(  \alpha_{0}\right)  +\frac{i\pi}{4}%
sgn\eta_{11}^{^{^{\prime\prime}}}\left(  \alpha_{0}\right)  \right)
\]
